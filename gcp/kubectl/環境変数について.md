# Volume
## ConfigMap
- podの中にファイル（ディスク）を用意してそこから情報をcatして読み取る
```
vim configMap.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: example-config
data:
  example.property.1: hello
  example.property.2: world
---
apiVersion: v1
kind: Pod
metadata:
  name: pod-using-config
spec:
  containers:
    - name: example-container
      image: busybox
      # あとからexecで操作するためsleepさせておく
      command: ["sleep", "3600"]
      # 定義したConfigMapのvolumeをマウントする
      volumeMounts:
        - name: config-volume
          mountPath: /etc/config
  volumes:
    # ConfigMapのvolume定義
    - name: config-volume
      configMap:
        # 作ったConfigMapのname
        name: example-config


kubectl apply -f configMap.yaml
kubectl exec pod-using-config -- cat /etc/config/example.property.1
kubectl exec pod-using-config -- cat /etc/config/example.property.2
```
## emptyDir
```
vim emptyDir.yaml

apiVersion: v1
kind: Pod
metadata:
  name: test-pd
spec:
  containers:
  - name: example-container
    image: busybox
      # あとからexecで中に入り操作するためsleepさせておく
    command: ["sleep", "3600"]
    volumeMounts:
    - mountPath: /cache
      name: cache-volume
  volumes:
  - name: cache-volume
    emptyDir: {}


kubectl apply -f emptyDir.yaml
kubectl exec -it  test-pd -- /bin/ash
touch /cache/hoge
ls /cache/
exit
kubectl delete pod test-pd
kubectl apply -f emptyDir.yaml
kubectl exec -it  test-pd -- /bin/ash
ls /cache/
```
